## 二、核心策略

### 2.1 减少堆分配

```go
// ❌ 高分配：产生GC压力
func processRequest(data []byte) []byte {
    result := make([]byte, len(data))  // 堆分配
    // 处理...
    return result
}

// ✅ 低分配：使用对象池
var bufferPool = sync.Pool{
    New: func() interface{} {
        return make([]byte, 4096)
    },
}

func processRequestOptimized(data []byte) []byte {
    buf := bufferPool.Get().([]byte)
    defer bufferPool.Put(buf)

    // 使用buf处理...
    result := buf[:len(data)]
    copy(result, data)
    return result
}
```

### 2.2 栈分配优化

```go
// ✅ 小数组栈分配
func smallBuffer() {
    var buf [256]byte  // 大概率栈分配
    use(buf[:])
}

// ✅ 避免逃逸
func noEscape() *int {
    x := 42
    return &x  // 逃逸到堆
}

func betterEscape() int {
    x := 42
    return x  // 不逃逸
}
```

### 2.3 预分配切片

```go
// ❌ 多次扩容
func collect(items []Item) []Result {
    var results []Result
    for _, item := range items {
        results = append(results, process(item))  // 可能多次扩容
    }
    return results
}

// ✅ 预分配
func collectOptimized(items []Item) []Result {
    results := make([]Result, 0, len(items))  // 预分配
    for _, item := range items {
        results = append(results, process(item))
    }
    return results
}
```

## 三、GC调优配置

### 3.1 环境变量

```bash
# 更激进的GC（更低延迟）
export GOGC=50

# 内存限制（容器环境）
export GOMEMLIMIT=6GiB

# GOMAXPROCS通常保持默认
```

### 3.2 运行时调优

```go
import (
    "runtime"
    "runtime/debug"
)

func init() {
    // 设置内存限制（软限制）
    debug.SetMemoryLimit(6 << 30)  // 6GB

    // 设置GC目标百分比
    debug.SetGCPercent(50)
}

// 请求前强制GC（极端场景）
func prepareForCriticalRequest() {
    runtime.GC()  // 立即执行GC
}
```

## 四、监控与诊断

### 4.1 GC指标收集

```go
import "runtime/metrics"

func collectGCMetrics() {
    samples := []metrics.Sample{
        {Name: "/gc/pause:seconds"},
        {Name: "/gc/heap/allocs:bytes"},
        {Name: "/gc/heap/frees:bytes"},
        {Name: "/gc/cycles/total:gc"},
    }

    metrics.Read(samples)

    for _, s := range samples {
        switch s.Name {
        case "/gc/pause:seconds":
            pause := s.Value.Float64()
            if pause > 0.001 {  // >1ms
                log.Printf("GC pause: %v", pause)
            }
        }
    }
}
```

### 4.2 延迟追踪

```go
func tracedHandler(w http.ResponseWriter, r *http.Request) {
    start := time.Now()

    // 处理请求...
    process(r)

    duration := time.Since(start)
    if duration > time.Millisecond {
        log.Printf("Slow request: %v", duration)
    }
}
```

## 五、实战案例

### 5.1 金融交易服务

```go
type TradingService struct {
    orderPool sync.Pool
}

func (s *TradingService) ProcessOrder(ctx context.Context, req *OrderRequest) (*OrderResponse, error) {
    // 从对象池获取
    order := s.orderPool.Get().(*Order)
    defer s.orderPool.Put(order)

    // 重置状态
    *order = Order{}
    order.ID = generateID()
    order.Amount = req.Amount
    order.Price = req.Price

    // 快速处理
    if err := s.validate(order); err != nil {
        return nil, err
    }

    if err := s.execute(order); err != nil {
        return nil, err
    }

    return &OrderResponse{OrderID: order.ID}, nil
}
```

### 5.2 游戏服务器

```go
type GameServer struct {
    statePool sync.Pool
    bufPool   sync.Pool
}

func (s *GameServer) Update(gameID string, input PlayerInput) error {
    // 使用预分配缓冲区
    buf := s.bufPool.Get().([]byte)
    defer s.bufPool.Put(buf)

    // 序列化
    n := serialize(input, buf)

    // 广播（零分配路径）
    s.broadcast(gameID, buf[:n])

    return nil
}
```

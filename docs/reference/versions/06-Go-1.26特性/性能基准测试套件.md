## 1. GC 性能测试

### gc_bench_test.go

```go
package main

import (
 "runtime"
 "testing"
 "time"
)

// BenchmarkGCPause 测试 GC 暂停时间
func BenchmarkGCPause(b *testing.B) {
 // 生成垃圾
 for i := 0; i < b.N; i++ {
  data := make([]byte, 10*1024*1024) // 10MB
  _ = data

  // 强制 GC
  start := time.Now()
  runtime.GC()
  pause := time.Since(start)

  b.ReportMetric(float64(pause.Microseconds()), "pause_us")
 }
}

// BenchmarkGCTotal 测试 GC 总开销
func BenchmarkGCTotal(b *testing.B) {
 var m1, m2 runtime.MemStats

 runtime.ReadMemStats(&m1)

 b.ResetTimer()
 for i := 0; i < b.N; i++ {
  // 分配内存
  data := make([][]byte, 100)
  for j := range data {
   data[j] = make([]byte, 1024*1024) // 1MB
  }
  _ = data
 }

 runtime.GC()
 runtime.ReadMemStats(&m2)

 // 报告指标
 b.ReportMetric(float64(m2.NumGC-m1.NumGC), "gc_count")
 b.ReportMetric(float64(m2.GCCPUFraction*100), "gc_cpu_percent")
}

// BenchmarkGCThroughput 测试 GC 吞吐量
func BenchmarkGCThroughput(b *testing.B) {
 b.ReportAllocs()

 for i := 0; i < b.N; i++ {
  // 大量小对象分配
  type Item struct {
   ID   int
   Data [1024]byte
  }

  items := make([]*Item, 10000)
  for j := range items {
   items[j] = &Item{ID: j}
  }
  _ = items
 }
}
```

### 运行测试

```bash
# 运行 GC 基准测试
go test -bench=BenchmarkGC -benchmem -count=5

# 对比两个版本
# Go 1.25
go1.25 test -bench=BenchmarkGC -count=5 > gc-old.txt

# Go 1.26
go1.26 test -bench=BenchmarkGC -count=5 > gc-new.txt

# 对比结果
benchstat gc-old.txt gc-new.txt
```

### 预期结果

```

GCPause-8     500µs ±10%     350µs ±8%      -30.00%
GCTotal-8     2.0s ±5%       1.6s ±4%       -20.00%

name          old alloc/op   new alloc/op   delta
```

```go
package main

import (
 "bytes"
 "io"
 "strings"
 "testing"

// BenchmarkReadAllSmall 测试小数据 io.ReadAll
func BenchmarkReadAllSmall(b *testing.B) {
 data := []byte(strings.Repeat("x", 1024)) // 1KB

 b.ResetTimer()

  _, err := io.ReadAll(r)

// BenchmarkReadAllMedium 测试中等数据 io.ReadAll

 b.ReportAllocs()
 b.ResetTimer()

 for i := 0; i < b.N; i++ {
  r := bytes.NewReader(data)
  _, err := io.ReadAll(r)
  if err != nil {
   b.Fatal(err)
  }
 }
}

// BenchmarkReadAllLarge 测试大数据 io.ReadAll
func BenchmarkReadAllLarge(b *testing.B) {
 data := []byte(strings.Repeat("x", 10*1024*1024)) // 10MB

 b.ReportAllocs()
 b.ResetTimer()

 for i := 0; i < b.N; i++ {
  r := bytes.NewReader(data)
  _, err := io.ReadAll(r)
  if err != nil {
   b.Fatal(err)
  }
 }
}

// BenchmarkReadAllChunked 测试分块数据
func BenchmarkReadAllChunked(b *testing.B) {
 // 模拟网络数据
 chunks := make([][]byte, 1000)
 for i := range chunks {
  chunks[i] = []byte(strings.Repeat("x", 1024))
 }

 b.ReportAllocs()
 b.ResetTimer()

 for i := 0; i < b.N; i++ {
  var buf bytes.Buffer
  for _, chunk := range chunks {
   buf.Write(chunk)
  }

  r := bytes.NewReader(buf.Bytes())
  _, err := io.ReadAll(r)
  if err != nil {
   b.Fatal(err)
  }
 }
}
```

## 4. CGO 性能测试

### cgo_bench_test.go

```go
package main

/*
#include <stdlib.h>
#include <string.h>

void process_data(char* data, int len) {
    for (int i = 0; i < len; i++) {
        data[i] = data[i] + 1;
    }
}

int compute(int n) {
    int sum = 0;
    for (int i = 0; i < n; i++) {
        sum += i;
    }
    return sum;
}
*/
import "C"
import (
 "testing"
 "unsafe"
)

// BenchmarkCGOSimple 简单 CGO 调用
func BenchmarkCGOSimple(b *testing.B) {
 for i := 0; i < b.N; i++ {
  result := C.compute(C.int(100))
  _ = result
 }
}

// BenchmarkCGOWithData 带数据的 CGO 调用
func BenchmarkCGOWithData(b *testing.B) {
 data := make([]byte, 1024)
 for i := range data {
  data[i] = byte(i)
 }

 b.ResetTimer()

 for i := 0; i < b.N; i++ {
  cData := (*C.char)(unsafe.Pointer(&data[0]))
  C.process_data(cData, C.int(len(data)))
 }
}

// BenchmarkCGOBatch 批量 CGO 调用
func BenchmarkCGOBatch(b *testing.B) {
 for i := 0; i < b.N; i++ {
  for j := 0; j < 100; j++ {
   C.compute(C.int(j))
  }
 }
}
```

### 运行测试

```bash
# 需要 CGO 启用
CGO_ENABLED=1 go test -bench=BenchmarkCGO -count=5
```

## 测试运行指南

### 一键运行所有测试

```bash
#!/bin/bash
# run_all_benchmarks.sh

echo "=== Go 1.26 性能基准测试套件 ==="
echo ""

# 保存结果
mkdir -p benchmark_results

# 运行所有测试
go test -bench=. -benchmem -count=5 2>&1 | tee benchmark_results/all.txt

# 生成摘要
echo ""
echo "=== 测试摘要 ==="
grep "^Benchmark" benchmark_results/all.txt | head -20

echo ""
echo "=== 内存分配 ==="
grep "allocs/op" benchmark_results/all.txt | head -20

echo ""
echo "结果已保存到 benchmark_results/all.txt"
```

### 对比两个版本

```bash
#!/bin/bash
# compare_versions.sh

VERSION_OLD=${1:-"go1.25"}
VERSION_NEW=${2:-"go1.26"}

echo "对比 $VERSION_OLD vs $VERSION_NEW"

# 新版本
```

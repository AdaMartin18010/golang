## 3. Custom Generators

```go
// mygenerator.go
package main

import (
    "fmt"
    "go/ast"
    "go/parser"
    "go/token"
    "os"
    "strings"
    "text/template"
)

func main() {
    if len(os.Args) < 2 {
        fmt.Println("Usage: mygenerator <file.go>")
        os.Exit(1)
    }

    filename := os.Args[1]

    // Parse source file
    fset := token.NewFileSet()
    f, err := parser.ParseFile(fset, filename, nil, parser.ParseComments)
    if err != nil {
        panic(err)
    }

    // Find structs with +gen marker
    var structs []string
    for _, decl := range f.Decls {
        if genDecl, ok := decl.(*ast.GenDecl); ok {
            if genDecl.Tok == token.TYPE {
                for _, spec := range genDecl.Specs {
                    if typeSpec, ok := spec.(*ast.TypeSpec); ok {
                        if _, ok := typeSpec.Type.(*ast.StructType); ok {
                            // Check for +gen comment
                            if hasMarker(genDecl.Doc, "+gen") {
                                structs = append(structs, typeSpec.Name.Name)
                            }
                        }
                    }
                }
            }
        }
    }

    // Generate code
    generateCode(structs)
}

func hasMarker(doc *ast.CommentGroup, marker string) bool {
    if doc == nil {
        return false
    }
    for _, comment := range doc.List {
        if strings.Contains(comment.Text, marker) {
            return true
        }
    }
    return false
}

func generateCode(structs []string) {
    tmpl := `// Code generated by mygenerator; DO NOT EDIT.
package {{.Package}}

{{range .Structs}}
func (s *{{.}}) Validate() error {
    // TODO: implement validation
    return nil
}
{{end}}
`

    t := template.Must(template.New("code").Parse(tmpl))
    t.Execute(os.Stdout, map[string]interface{}{
        "Package": "mypackage",
        "Structs": structs,
    })
}
```

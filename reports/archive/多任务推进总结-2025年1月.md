```go
// 泛型Worker Pool实现
type WorkerPool[T any] struct {
    workers    int
    jobQueue   chan Job[T]
    resultChan chan Result[T]
    wg         sync.WaitGroup
}

// 类型安全的任务处理
func (wp *WorkerPool[T]) Submit(job Job[T]) error {
    select {
    case wp.jobQueue <- job:
        return nil
    case <-wp.ctx.Done():
        return wp.ctx.Err()
    default:
        return fmt.Errorf("job queue is full")
    }
}
```

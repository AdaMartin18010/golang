### 8.1.2.1 **GitHub Actions**

```yaml
name: Go CI
on: [push, pull_request]
jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - name: Set up Go
        uses: actions/setup-go@v4
        with:
          go-version: '1.22'
      - name: Test
        run: go test -v ./...
      - name: Coverage
        run: go test -coverprofile=cover.out ./... && go tool cover -func=cover.out

```
